class Solution {
    public int removeDuplicates(int[] nums) {
        int i =0;
        for (int j = 1; j < nums.length; j++) {
            if (nums[j] != nums[i]) {
                i++;
                nums[i]=nums[j];
            }
            
        }
       return  i+1;
    }
}
//using inplace without creating a new array and not using arraylist concept here
